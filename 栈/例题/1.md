## 判断一个表达式中的括号是否配对，表达式已经存入字符数组exp[]中,表达式中的字符个数为n;

```
#include "Lstack.h"
#include "utils.h"
#include <iostream>
int match(LStack &stack, char test[]) {
	while (*test != '\0') {
		if (*test == '(') {
			stack.push('(');
		}
		else if (*test == ')') {
			char top = stack.get_value();
			if (top == ')' || top == '$') {
				return -1;
			}
			else if (top == '('){
				stack.pop();
			}
		}
		test++;
	}
	if (stack.get_value() == '$') {
		return 1;
	}
	else {
		return -1;
	}
}
int main()
{
	char test[] = { "1+2*(2-1)()+1+3+()(" };
	LStack stack = LStack();
	cout<<match(stack, test);
}
```
